int binarySearch(List list, bool matches(item)) #

Find the first entry in a sorted list that matches a monotonic predicate. Given a result n, that all items before n will not match, n matches, and all items after n match too. The result is -1 when there are no items, 0 when all items match, and list.length when none does.

int binarySearch(List list, bool matches(item)) {
 if (list.length == 0) return -1;
 if (matches(list.first)) return 0;
 if (!matches(list.last)) return list.length;

 int min = 0;
 int max = list.length - 1;
 while (min < max) {
   var half = min + ((max - min) ~/ 2);
   if (matches(list[half])) {
     max = half;
   } else {
     min = half + 1;
   }
 }
 return max;
}

 void _addOuterHtml(StringBuffer str) {
   // Don't escape text for certain elements, notably <script>.
   if (rcdataElements.contains(parent.tagName) ||
       parent.tagName == 'plaintext') {
     str.write(value);
   } else {
     str.write(htmlSerializeEscape(value));
   }
 }

 }
}
